## [75. 颜色分类](https://leetcode-cn.com/problems/sort-colors/)

三色问题, 荷兰国旗问题

```javascript
/**
 * @param {number[]} nums
 * @return {void} Do not return anything, modify nums in-place instead.
 */
var sortColors = function(nums) {
    if(!nums || nums.length < 2){
      return nums
    }
    const swap = (i, j) => {
      const tmp = nums[i]
      nums[i] = nums[j]
      nums[j] = tmp
    }

    let left = mid = 0,
        right = nums.length - 1
    while(mid <= right){
      if(nums[mid] == 0){
        if(left == mid){
          left += 1
          mid += 1
        }else{
          swap(left, mid)
          left += 1
        }
      }else if(nums[mid] == 2){
        swap(mid, right)
        right -= 1
      }else{
        mid += 1
      }
    }
};
```
